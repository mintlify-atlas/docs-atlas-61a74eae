---
title: 'Response'
description: 'HTTP Response API reference for sending responses in AdonisJS'
---

The `Response` class provides methods for sending HTTP responses, setting headers, cookies, and managing response data.

## Importing

```typescript
import { HttpContext } from '@adonisjs/core/http'

export default class UsersController {
  async index({ response }: HttpContext) {
    // Access response
  }
}
```

## Properties

<ParamField path="finished" type="boolean">
  Whether the response has been sent
</ParamField>

<ParamField path="isPending" type="boolean">
  Whether the response is pending
</ParamField>

<ParamField path="headersSent" type="boolean">
  Whether headers have been sent
</ParamField>

## Sending Responses

### send()

Send a response body.

```typescript
response.send('Hello World')
response.send({ message: 'Success' })
response.send(Buffer.from('data'))
```

<ParamField path="body" type="any">
  Response body (string, object, buffer, stream)
</ParamField>

**Returns:** `void`

### json()

Send a JSON response.

```typescript
response.json({ 
  users: [],
  meta: { total: 0 }
})
```

<ParamField path="body" type="any">
  Data to serialize as JSON
</ParamField>

**Returns:** `void`

### jsonp()

Send a JSONP response.

```typescript
response.jsonp({ data: [] })
```

<ParamField path="body" type="any">
  Data to serialize as JSONP
</ParamField>

<ParamField path="callbackName" type="string" default="'callback'">
  JSONP callback function name
</ParamField>

**Returns:** `void`

### stream()

Stream a response.

```typescript
const fileStream = fs.createReadStream('video.mp4')
response.stream(fileStream)
```

<ParamField path="stream" type="ReadableStream">
  Readable stream to pipe
</ParamField>

**Returns:** `void`

### download()

Download a file.

```typescript
response.download('storage/invoices/invoice-123.pdf')
```

<ParamField path="filePath" type="string">
  Path to the file
</ParamField>

<ParamField path="fileName" type="string">
  Optional filename for download
</ParamField>

**Returns:** `void`

### attachment()

Download a file with a custom name.

```typescript

  'storage/reports/report-2024.pdf',
  'Annual Report.pdf'
)
```

<ParamField path="filePath" type="string">
  Path to the file
</ParamField>

<ParamField path="fileName" type="string">
  Filename for download
</ParamField>

**Returns:** `void`

## Example Usage

```typescript
import { HttpContext } from '@adonisjs/core/http'

export default class UsersController {
  async index({ response }: HttpContext) {
    const users = await User.all()
    
    // JSON response
    return response.json({
      data: users,
      meta: { total: users.length }
    })
  }
  
  async store({ request, response }: HttpContext) {
    const user = await User.create(request.all())
    
    // Created response with location header
    return response
      .status(201)
      .header('Location', `/users/${user.id}`)
      .json({ user })
  }
  
  async download({ response }: HttpContext) {
    // File download
    return response.download('storage/reports/report.pdf')
  }
  
  async redirect({ response }: HttpContext) {
    // Redirect with flash message
    return response
      .redirect()
      .withQs({ success: 'User created' })
      .toRoute('users.index')
  }
}
```
